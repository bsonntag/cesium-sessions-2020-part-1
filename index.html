The Web

Source: MDN
Hoje vamos falar sobre JavaScript, mas antes vamos falar primeiro um pouco sobre como a Web funciona.
Um website é composto por vários ficheiros: ficheiros HTML, CSS, JavaScript, imagens, etc.
Estes ficheiros estão alojados num servidor web, que os serve através do protocolo HTTP.
Quando alguém quer aceder ao vosso website, terá que abrir um browser e introduzir o endereço do vosso website, ou navegar até lá a partir de um motor de pesquisa como o Google.
Ao abrir o website, o browser vai fazer vários pedidos HTTP para pedir os vários ficheiros que compoem o site. Esses ficheiros serão então enviados para o browser em resposta a esses pedidos.

Destructuring, Rest and Spread

Destructuring assignment

Makes it possible to unpack values from arrays,
or properties from objects, into distinct variables.

From MDN
Array destructuring

let colors = ['red', 'green', 'blue'];

let firstColor = colors[0];
let secondColor = colors[1];

console.log(firstColor); // red
console.log(secondColor); // green
              

let colors = ['red', 'green', 'blue'];

let [firstColor, secondColor] = colors;

console.log(firstColor); // red
console.log(secondColor); // green
Object destructuring

let user = {
  firstName: 'Benjamim',
  lastName: 'Sonntag',
  github: 'bsonntag'
};

let firstName = user.firstName;
let lastName = user.lastName;

console.log(firstName); // Benjamim
console.log(lastName); // Sonntag
              

let user = {
  firstName: 'Benjamim',
  lastName: 'Sonntag',
  github: 'bsonntag'
};

let { firstName, lastName } = user;

console.log(firstName); // Benjamim
console.log(lastName); // Sonntag
Default values

let [firstColor = 'purple'] = colors;

let { firstName = 'John' } = user;
Assigning to a different name

let { firstName: userName } = user;

console.log(userName); // Benjamim
Destructuring function results

let [firstUser, secondUser] = getAllUsers();

let { github } = getMostActiveUser();
Destructuring function arguments

function getUserFullName({ firstName, lastName }) {
  return firstName + ' ' + lastName;
}
Assigning the rest

let colors = ['red', 'green', 'blue'];

let [firstColor, ...rest] = colors;

console.log(firstColor); // red
console.log(rest); // ['green', 'blue']
              

let user = {
  firstName: 'Benjamim',
  lastName: 'Sonntag',
  github: 'bsonntag'
};

let { github, ...rest } = user;

console.log(github); // bsonntag
console.log(rest);
// { firstName: 'Benjamim', lastName: 'Sonntag' }

Function rest parameters


function sum(...values) {
  let result = 0;
  for (let value of values) {
    result += value;
  }
  return result;
}

Spread operator

  • Expands arrays into other arrays or function arguments
  • Expands objects into other objects
Spreading arrays into other arrays

let colors = ['red', 'green', 'blue'];

let moreColors = ['purple', 'yellow', ...colors];

console.log(moreColors);
// ['purple', 'yellow', 'red', 'green', 'blue']
              

let colors = ['red', 'green', 'blue'];

let moreColors = ['purple', ...colors, 'yellow'];

console.log(moreColors);
// ['purple', 'red', 'green', 'blue', 'yellow']
Spreading arrays into function arguments

let numbers = [2, 4, 8, 16];

let max = Math.max(...numbers);

console.log(max); // 16
Spreading objects into objects

let user = { name: 'Benjamim Sonntag', role: 'user' };

let superUser = {
  ...user,
  role: 'admin'
};

console.log(superUser);
// { name: 'Benjamim Sonntag', role: 'admin' }
              

let user = { name: 'Benjamim Sonntag', role: 'user' };

let superUser = {
  role: 'admin',
  ...user // user.role will overwrite the role. ⚠️
};

console.log(superUser);
// { name: 'Benjamim Sonntag', role: 'user' } ⚠️

Optional chaining

Permits reading the value of a property located deep within a chain of connected objects without having to expressly validate that each reference in the chain is valid.

O operador de optional chaining permite-nos ler o valor de uma propriedade aninhada numa sequência de objetos dentro de objetos, sem ser preciso validar que cada um desses objectos existe. 

let user = {
  account: {
    name: 'Benjamim'
  }
};
            

function getUserName(user) {
  return user.account.name;
}
            

function getUserName(user) {
  if (user != null) {
    return user.account.name;
  }
}
            

function getUserName(user) {
  return user?.account.name;
}

Promises and Async-await

Promise

Represents the eventual completion (or failure) of an asynchronous operation and its resulting value.

Promise states
  • Pending
  • Resolved
  • Rejected

let usersPromise = fetchCurrentUser();

usersPromise.then(user => {
  console.log(user);
});

usersPromise.catch(error => {
  console.log(error);
});
            

fetchCurrentUser()
  .then(user => {
    console.log(user);
  })
  .catch(error => {
    console.log(error);
  });
            

fetchCurrentUser()
  .then(user => {
    return fetchUserBlogPosts(user);
  })
  .then(blogPosts => {
    console.log(blogPosts);
  });

Async-await

The await operator can be used to wait for a promise. It can only be used inside an async function. 


async function getUserAndBlogPosts() {
  try {
    let user = await fetchCurrentUser();
    console.log(user);

    let blogPosts = await fetchUserBlogPosts(user);
    console.log(blogPosts);
  } catch (error) (
    console.log(error);
  }
}
